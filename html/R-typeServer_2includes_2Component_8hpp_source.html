301 void one_shot(registry &reg, std::shared_ptr<network::INetworker> &_net,
302 registry::entity_t parent, int msl_type, float x, float y, float vx, float vy, float ax, float ay)
303 {
304 registry::entity_t obj = reg.entity_manager.spawn_entity();
305 std::size_t room_id = reg.get_components<component::room_id>()[parent].value().id;
306
307 reg.entity_manager.emplace_component<component::EntityType>(obj, component::Missile);
308 reg.get_components<component::room_info>()[room_id].value().add_entity(reg, obj);
309 reg.entity_manager.emplace_component<component::parent>(obj, parent);
310 reg.entity_manager.emplace_component<component::damage>(obj, damage_val);
311 reg.entity_manager.emplace_component<component::health>(obj, 1);
312 reg.entity_manager.emplace_component<component::position>(obj, x, y);
313 reg.entity_manager.emplace_component<component::velocity>(obj, vx, vy);
314 reg.entity_manager.emplace_component<component::acceleration>(obj, ax, ay);
315 reg.entity_manager.emplace_component<component::area>(obj, x, y, 30, 30,
316 [&, _net, parent, obj, room_id, msl_type](registry &reg, registry::entity_t oth) {
317 auto &area = reg.get_components<component::area>()[obj].value();
318
319 if (area.collided.find(oth) == area.collided.end())
320 area.collided.emplace(oth, ClockTime());
321 if (area.collided.at(oth).launched() &&
322 area.collided.at(oth).getElapsedTime<double>() < 2)
323 return;
324 if (!reg.get_components<component::EntityType>().has_value_at(oth))
325 return;
326
327 auto &oth_type = reg.get_components<component::EntityType>()[oth].value();
328 if (oth_type != adversary_type)
329 return;
330
331 auto &health = reg.get_components<component::health>()[obj].value();
332 auto &oth_health = reg.get_components<component::health>()[oth].value();
333 auto &oth_score = reg.get_components<component::score>()[oth].value();
334 auto &parent_type = reg.get_components<component::EntityType>()[parent].value();
335 if (oth_health.val) {
336 if (!reg.entity_manager.has_components<component::shield>(oth))
337 oth_health.val -= damage_val;
338 if (msl_type == 2)
339 oth_health.val = 0;
340 if (msl_type != 2)
341 health.val = 0;
342 if (oth_health.val == 0) {
343 if (parent_type == component::Player) {
344 auto &parent_score = reg.get_components<component::score>()[parent].value();
345 auto &parent_net_id = reg.get_components<component::network_id>()[parent].value().id;
346 parent_score.val += oth_score.val;
347 network::datatrans data(network::UpdateScore,
348 network::entity_info(parent, 0, 0, parent_score.val));
349 _net->set_peer_id(parent_net_id);
350 *_net << data;
351 }
352 alert_clients(reg, _net, room_id, {}, network::EntityKilled,
353 network::entity_info(), Vector2D(area.x, area.y));
354 }
355 if (oth_type == component::Player) {
356 auto &oth_net_id = reg.get_components<component::network_id>()[oth].value().id;
357 network::datatrans data(network::UpdateHealth,
358 network::entity_info(oth, oth_health.val));
359 _net->set_peer_id(oth_net_id);
360 *_net << data;
361 alert_clients(reg, _net, room_id, {}, network::PlayerTouched);
362 }
363 if (!area.collided.at(oth).launched())
364 area.collided.at(oth).launchChrono();
365 else
366 area.collided.at(oth).reset();
367 std::cout << "Entity " << parent << " adversary touched" << std::endl;
368 }
369 });
370

372
373 alert_clients(reg, _net, room_id, {}, network::CreateMissile,
374 network::entity_info(obj, 1, msl_type), Vector2D(x, y), Vector2D(vx, vy), Vector2D(ax, ay));

376 }
