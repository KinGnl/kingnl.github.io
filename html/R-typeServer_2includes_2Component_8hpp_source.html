240 void one_shot(registry &reg, std::shared_ptr<network::INetworker> &_net,
241 registry::entity_t parent, float x, float y, float vx, float vy, float ax, float ay)
242 {
243 registry::entity_t obj = reg.entity_manager.spawn_entity();
244 std::size_t room_id = reg.get_components<component::room_id>()[parent].value().id;
245
246 reg.entity_manager.emplace_component<component::EntityType>(obj, component::Missile);
247 reg.entity_manager.emplace_component<component::room_id>(obj, room_id);
248 reg.get_components<component::room_info>()[room_id].value().add_entity(obj);
249 reg.entity_manager.emplace_component<component::parent>(obj, parent);
250 reg.entity_manager.emplace_component<component::damage>(obj, damage_val);
251 reg.entity_manager.emplace_component<component::health>(obj, 1);
252 reg.entity_manager.emplace_component<component::position>(obj, x, y);
253 reg.entity_manager.emplace_component<component::velocity>(obj, vx, vy);
254 reg.entity_manager.emplace_component<component::acceleration>(obj, ax, ay);
255 reg.entity_manager.emplace_component<component::area>(obj, x, y, 32, 32,
256 [&, _net, parent, obj](registry &reg, registry::entity_t oth) {
257 auto &area = reg.get_components<component::area>()[obj].value();
258
259 if (area.collided.find(oth) == area.collided.end())
260 area.collided.emplace(oth, ClockTime());
261 if (area.collided.at(oth).launched() &&
262 area.collided.at(oth).getElapsedTime<double>() < 2)
263 return;
264 if (!reg.get_components<component::EntityType>().has_value_at(oth))
265 return;
266
267 auto &oth_type = reg.get_components<component::EntityType>()[oth].value();
268 if (oth_type != adversary_type)
269 return;
270
271 auto &oth_health = reg.get_components<component::health>()[oth].value();
272 auto &oth_score = reg.get_components<component::score>()[oth].value();
273 auto &parent_type = reg.get_components<component::EntityType>()[parent].value();
274 if (oth_health.val) {
275 oth_health.val -= damage_val;
276 if (parent_type == component::Player) {
277 auto &parent_score = reg.get_components<component::score>()[parent].value();
278 auto &parent_net_id = reg.get_components<component::network_id>()[parent].value();
279 parent_score.val += oth_score.val;
280 network::datatrans data(network::UpdateScore,
281 network::entity_info(parent, 0, 0, parent_score.val));
282 _net->set_peer_id(parent_net_id.id);
283 *_net << data;

286 auto &oth_net_id = reg.get_components<component::network_id>()[oth].value().id;
287 network::datatrans data(network::UpdateHealth,

292 if (!area.collided.at(oth).launched())

295 area.collided.at(oth).reset();
296 std::cout << "Adversary touched" << std::endl;
297 }
298 });
299 alert_clients(reg, _net, room_id, {}, network::CreateMissile,
300 network::entity_info(obj), Vector2D(x, y), Vector2D(vx, vy), Vector2D(ax, ay));
301 }
